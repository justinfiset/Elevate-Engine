125 Mesh Mesh::GenerateUVSphere(float radius, int latitudes, int longitudes)
126 {
127 if (longitudes < 3) longitudes = 3;
128 if (latitudes < 2) latitudes = 2;
129
130 const float pi = std::acos(-1.0f);
131 std::vector<Vertex> vertices;
132 std::vector<uint32_t> indices;
133
134 float latStep = pi / latitudes;
135 float longStep = 2 * pi / longitudes;
136
137 for (int lat = 0; lat <= latitudes; ++lat)
138 {
139 float phi = (pi / 2) - (lat * latStep); // From 90* down (starting from north pole basicly)
140 const float xy = radius * cosf(phi);
141 const float z = radius * sinf(phi);
142
143 for (int lon = 0; lon <= longitudes; ++lon)
144 {
145 float theta = lon * longStep;
146
147 Vertex v;
148 v.Position = {
149 xy * cosf(theta),
150 xy * sinf(theta),
151 z
152 };
153 v.TexCoords = {
154 (float) lon / longitudes,
155 (float) lat / latitudes
156 };
157 v.Normal = glm::normalize(v.Position);
158 vertices.push_back(v);
159 }
160 }
161
162 for (int lat = 0; lat < latitudes; ++lat)
163 {
164 for (int lon = 0; lon < longitudes; ++lon)
165 {
166 const uint32_t curr = lat * (longitudes + 1) + lon;
167 const uint32_t next = curr + longitudes + 1;
168
169 indices.insert(indices.end(), {
170 curr, next, curr + 1,
171 next, next + 1, curr + 1
172 });
173 }
174 }
175
176 return Mesh(vertices, indices, {});

189 Mesh Mesh::GeneratePlane(float size, int resolution) {

192
193 float step = size / resolution;
194 float offset = size * 0.5f;
195
196 for (int i = 0; i <= resolution; i++) {
197 for (int j = 0; j <= resolution; j++) {
198 Vertex v;
199 v.Position = { j * step - offset, 0.0f, i * step - offset };
200 v.Normal = { 0.0f, 1.0f, 0.0f };
201 v.TexCoords = { (float)j / resolution, (float)i / resolution };
202 vertices.push_back(v);
203 }
204 }
205
206 for (int i = 0; i < resolution; i++) {
207 for (int j = 0; j < resolution; j++) {
208 int row1 = i * (resolution + 1);
